def count_ways_as_sum_of_two_squares(n):
    ways = 0
    for x in range(int(n**0.5) + 1):
        y = (n - x**2)**0.5
        if y.is_integer():
            ways += 1
    return ways

# Test cases
for num in [3, 4, 5, 6]:
    ways = count_ways_as_sum_of_two_squares(num)
    print(f"Number of ways to write {num} as a sum of two squares: {ways}")
